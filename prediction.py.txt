import json

def predict_temperature():
    try:
        with open("sensors.json", "r") as f:
             sensor_data = json.load(f)
        data = list(sensor_data.values())

        if len(data) == 0:
            return {"prediction": "No data", "alert": "No Prediction"}

        # simple average prediction
        avg = sum(data) / len(data)

        temp = round(avg, 2)

        if temp > 25:
            alert = "🔥 HIGH TEMPERATURE ALERT"
        elif temp < 10:
            alert = "❄️ LOW TEMPERATURE ALERT"
        else:
            alert = "✅ Normal Temperature"

        return {
            "prediction": temp,
            "alert": alert
        }

    except:
        return {"prediction": "Error", "alert": "Check data"}
