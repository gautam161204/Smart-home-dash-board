import json
import numpy as np
from sklearn.preprocessing import MinMaxScaler
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, LSTM

def predict_temperature():

    # Load temperature history
    with open("temp_history.json", "r") as f:
        temps = json.load(f)

    # Need minimum data
    if len(temps) < 10:
        return {
            "prediction": "Need minimum 10 values",
            "alert": "No Prediction"
        }

    dataset = np.array(temps).reshape(-1,1)

    scaler = MinMaxScaler(feature_range=(0,1))
    scaled = scaler.fit_transform(dataset)

    X = []
    y = []

    for i in range(3, len(scaled)):
        X.append(scaled[i-3:i,0])
        y.append(scaled[i,0])

    X = np.array(X)
    y = np.array(y)

    X = np.reshape(X, (X.shape[0], X.shape[1],1))

    # Build model
    model = Sequential()
    model.add(LSTM(50, input_shape=(3,1)))
    model.add(Dense(1))

    model.compile(optimizer='adam', loss='mean_squared_error')

    model.fit(X, y, epochs=20, batch_size=1, verbose=0)

    # Predict next temp
    last3 = scaled[-3:]
    test = np.reshape(last3, (1,3,1))

    pred = model.predict(test, verbose=0)

    result = scaler.inverse_transform(pred)

    temp = round(float(result[0][0]),2)

    alert = "Normal Temperature ✅"
    if temp > 45:
        alert = "HIGH TEMPERATURE ALERT 🚨"

    return {
        "prediction": temp,
        "alert": alert
    }
